 30     var /**@const
 31            @type{!string}*/
 32         fons = odf.Namespaces.fons,
 33         /**@const
 34            @type{!string}*/
 35         stylens = odf.Namespaces.stylens,
 36         /**@const
 37            @type{!string}*/
 38         textns = odf.Namespaces.textns,
 39         /**@const
 40            @type{!Object.<string,string>}*/
 41         stylemap = {
 42             '1': 'decimal',
 43             'a': 'lower-latin',
 44             'A': 'upper-latin',
 45             'i': 'lower-roman',
 46             'I': 'upper-roman'
 47         };

 49     /**
 50      * @constructor
 51      */
 52     odf.ListStyleToCss = function ListStyleToCss() {
 53 
 54         var cssUnits = new core.CSSUnits(),
 55             odfUtils = new odf.OdfUtils();
 56 
 57         /**
 58          * Takes a value with a valid CSS unit and converts it to a CSS pixel value
 59          * @param {!string} value
 60          * @return {!number}
 61          */
 62         function convertToPxValue(value) {
 63             var parsedLength = odfUtils.parseLength(value);
 64             if (!parsedLength) {
 65                 runtime.log("Could not parse value '"+value+"'.");
 66                 // Return 0 as fallback, might have least bad results if used
 67                 return 0;
 68             }
 69             return cssUnits.convert(parsedLength.value, parsedLength.unit, "px");
 70         }
 71 
 72         /**
 73          * Appends the rule into the stylesheets and logs any errors that occur
 74          * @param {!CSSStyleSheet} styleSheet
 75          * @param {!string} rule
 76          */
 77         function appendRule(styleSheet, rule) {
 78             try {
 79                 styleSheet.insertRule(rule, styleSheet.cssRules.length);
 80             } catch (/**@type{!DOMException}*/e) {
 81                 runtime.log("cannot load rule: " + rule + " - " + e);
 82             }
 83         }
 84 
 85         /**
 86          * Return the supplied value with any backslashes escaped, and double-quotes escaped
 87          * @param {!string} value
 88          * @return {!string}
 89          */
 90         function escapeCSSString(value) {
 91             return value.replace(/\\/g, "\\\\").replace(/"/g, "\\\"");
 92         }
 93 
 94         /**
 95          * Gets the CSS content for a numbered list
 96          * @param {!Element} node
 97          * @return {!string}
 98          */
 99         function getNumberRule(node) {
100             var style = node.getAttributeNS(stylens, "num-format"),
101                 /**@type{!string}*/
102                 suffix = node.getAttributeNS(stylens, "num-suffix") || "",
103                 /**@type{!string}*/
104                 prefix = node.getAttributeNS(stylens, "num-prefix") || "",
105                 /**@type{!string}*/
106                 content = "";
107             if (prefix) {
108                 // Content needs to be on a new line if it contains slashes due to a bug in older versions of webkit
109                 // E.g., the one used in the qt runtime tests - https://bugs.webkit.org/show_bug.cgi?id=35010
110                 content += '"' + escapeCSSString(prefix) + '"\n';
111             }
112             if (stylemap.hasOwnProperty(style)) {
113                 content += " counter(list, " + stylemap[style] + ")";
114             } else if (style) {
115                 content += ' "' + style + '"';
116             } else {
117                 content += ' ""';
118             }
119             return 'content:' + content + ' "' + escapeCSSString(suffix) + '"';
120         }
121 
122         /**
123          * Gets the CSS content for a image bullet list
124          * @return {!string}
125          */
126         function getImageRule() {
127             return "content: none";
128         }
129 
130         /**
131          * Gets the CSS content for a bullet list
132          * @param {!Element} node
133          * @return {!string}
134          */
135         function getBulletRule(node) {
136             var bulletChar = node.getAttributeNS(textns, "bullet-char");
137             return 'content: "' + escapeCSSString(bulletChar) + '"';
138         }
139 
140         /**
141          * In label-width-and-position mode of specifying list layout the margin and indent specified in
142          * the paragraph style is additive to the layout specified in the list style.
143          *
144          *   fo:margin-left    text:space-before    fo:text-indent  +-----------+
145          * +---------------->+------------------>+----------------->|   label   |     LIST TEXT
146          *                                                          +-----------+
147          * +---------------->+------------------>+-------------------->LIST TEXT LIST TEXT LIST TEXT
148          *                                        text:min-label-width
149          *
150          * To get this additive behaviour we calculate an offset from the left side of the page which is
151          * the space-before +  min-label-width. We then apply this offset to each text:list-item
152          * element and apply the negative value of the offset to each text:list element. This allows the positioning
153          * provided in the list style to apply relative to the paragraph style as we desired. Then on each
154          * ::before pseudo-element which holds the label we apply the negative value of the min-label-width to complete
155          * the alignment from the left side of the page. We then apply the min-label-distance as padding to the right
156          * of the ::before psuedo-element to complete the list label placement.
157          *
158          * For the label-alignment mode the paragraph style overrides the list style but we specify offsets for
159          * the text:list and text:list-item elements to keep the code consistent between the modes
160          *
161          * Diagram and implementation based on: https://wiki.openoffice.org/wiki/Number_layout
162          *
163          * @param {!CSSStyleSheet} styleSheet
164          * @param {!string} name
165          * @param {!Element} node
166          * @param {!string} itemRule
167          * @return {undefined}
168          */
169         function addListStyleRule(styleSheet, name, node, itemRule) {
170             var selector = 'text|list[text|style-name="' + name + '"]',
171                 level = node.getAttributeNS(textns, "level"),
172                 listItemRule,
173                 listLevelProps = /**@type{!Element}*/(node.getElementsByTagNameNS(stylens, "list-level-properties")[0]),
174                 listLevelPositionSpaceMode = listLevelProps.getAttributeNS(textns, "list-level-position-and-space-mode"),
175                 listLevelLabelAlign = /**@type{!Element}*/(listLevelProps.getElementsByTagNameNS(stylens, "list-level-label-alignment")[0]),
176                 listIndent,
177                 textAlign,
178                 bulletWidth,
179                 labelDistance,
180                 bulletIndent,
181                 followedBy,
182                 leftOffset;
183 
184             // calculate CSS selector based on list level
185             level = level && parseInt(level, 10);
186             while (level > 1) {
187                 selector += ' > text|list-item > text|list';
188                 level -= 1;
189             }
190 
191             // TODO: fo:text-align is only an optional attribute with <style:list-level-properties>,
192             // needs to be found what should be done if not present. For now falling back to "left"
193             textAlign = listLevelProps.getAttributeNS(fons, "text-align") || "left";
194             // convert the start and end text alignments to left and right as
195             // IE does not support the start and end values for text alignment
196             switch (textAlign) {
197                 case "end":
198                     textAlign = "right";
199                     break;
200                 case "start":
201                     textAlign = "left";
202                     break;
203             }
204 
205             // get relevant properties from the style based on the list label positioning mode
206             if (listLevelPositionSpaceMode === "label-alignment") {
207                 // TODO: fetch the margin and indent from the paragraph style if it is defined there
208                 // http://docs.oasis-open.org/office/v1.2/os/OpenDocument-v1.2-os-part1.html#element-style_list-level-label-alignment
209                 // for now just fallback to "0px" if not defined on <style:list-level-label-alignment>
210                 listIndent = listLevelLabelAlign.getAttributeNS(fons, "margin-left") || "0px";
211                 bulletIndent = listLevelLabelAlign.getAttributeNS(fons, "text-indent") || "0px";
212                 followedBy = listLevelLabelAlign.getAttributeNS(textns, "label-followed-by");
213                 leftOffset = convertToPxValue(listIndent);
214 
215             } else {
216                 // this block is entered if list-level-position-and-space-mode
217                 // has the value label-width-and-position or is not present
218                 // TODO: fallback values should be read from parent styles or (system) defaults
219                 listIndent = listLevelProps.getAttributeNS(textns, "space-before") || "0px";
220                 bulletWidth = listLevelProps.getAttributeNS(textns, "min-label-width") || "0px";
221                 labelDistance = listLevelProps.getAttributeNS(textns, "min-label-distance") || "0px";
222                 leftOffset = convertToPxValue(listIndent) + convertToPxValue(bulletWidth);
223             }
224 
225             listItemRule = selector + ' > text|list-item';
226             listItemRule += '{';
227             listItemRule += 'margin-left: ' + leftOffset + 'px;';
228             listItemRule += '}';
229             appendRule(styleSheet, listItemRule);
230 
231             listItemRule = selector + ' > text|list-item > text|list';
232             listItemRule += '{';
233             listItemRule += 'margin-left: ' + (-leftOffset) + 'px;';
234             listItemRule += '}';
235             appendRule(styleSheet, listItemRule);
236 
237             // insert the list label before every immediate child of the list-item, except for lists
238             listItemRule = selector + ' > text|list-item > *:not(text|list):first-child:before';
239             listItemRule += '{';
240             listItemRule += 'text-align: ' + textAlign + ';';
241             listItemRule += 'counter-increment:list;';
242             listItemRule += 'display: inline-block;';
243 
244             if (listLevelPositionSpaceMode === "label-alignment") {
245                 listItemRule += 'margin-left: ' + bulletIndent + ';';
246                 if (followedBy === "space") {
247                     itemRule += ' \'\\a0\'';
248                 } else if (followedBy === "listtab") {
249                     // TODO: remove this padding once text:label-followed-by="listtab" is implemented
250                     // http://docs.oasis-open.org/office/v1.2/os/OpenDocument-v1.2-os-part1.html#attribute-text_label-followed-by
251                     listItemRule += 'padding-right: 0.2cm;';
252                 }
253             } else {
254                 listItemRule += 'min-width: ' + bulletWidth + ';';
255                 listItemRule += 'margin-left: -' + bulletWidth + ';';
256                 listItemRule += 'padding-right: ' + labelDistance + ';';
257             }
258             // Content needs to be on a new line if it contains slashes due to a bug in older versions of webkit
259             // E.g., the one used in the qt runtime tests - https://bugs.webkit.org/show_bug.cgi?id=35010
260             listItemRule += "\n" + itemRule + ';\n';
261             listItemRule += '}';
262             appendRule(styleSheet, listItemRule);
263         }
264 
265         /**
266          * Adds a CSS rule for every ODF list style
267          * @param {!CSSStyleSheet} styleSheet
268          * @param {!string} name
269          * @param {!Element} node
270          * @return {undefined}
271          */
272         function addRule(styleSheet, name, node) {
273             var n = node.firstChild, e, itemrule;
274             while (n) {
275                 if (n.namespaceURI === textns) {
276                     e = /**@type{!Element}*/(n);
277                     if (n.localName === "list-level-style-number") {
278                         itemrule = getNumberRule(e);
279                         addListStyleRule(styleSheet, name, e, itemrule);
280                     } else if (n.localName === "list-level-style-image") {
281                         itemrule = getImageRule();
282                         addListStyleRule(styleSheet, name, e, itemrule);
283                     } else if (n.localName === "list-level-style-bullet") {
284                         itemrule = getBulletRule(e);
285                         addListStyleRule(styleSheet, name, e, itemrule);
286                     }
287                 }
288                 n = n.nextSibling;
289             }
290         }
291 
292         /**
293          * Creates CSS styles from the given ODF list styles and applies them to the stylesheet
294          * @param {!CSSStyleSheet} stylesheet
295          * @param {!odf.StyleTree.Tree} styleTree
296          * @return {undefined}
297          */
298         this.applyListStyles = function (stylesheet, styleTree) {
299             var styleFamilyTree,
300                 node;
301 
302             /*jslint sub:true*/
303             // The available families are defined in StyleUtils.familyNamespacePrefixes.
304             styleFamilyTree = (styleTree["list"]);
305             /*jslint sub:false*/
306             if (styleFamilyTree) {
307                 Object.keys(styleFamilyTree).forEach(function (styleName) {
308                     node = /**@type{!odf.StyleTreeNode}*/(styleFamilyTree[styleName]);
309                     addRule(stylesheet, styleName, node.element);
310                 });
311             }
312         };
313     };
314 }());
